#include <iostream>

Cấu trúc rẽ nhánh

Câu lệnh if

if (⟨biểu thức điều kiện⟩)
{
    ⟨khối lệnh⟩
}

Nếu ⟨biểu thức điều kiện⟩ đúng thì thực hiện ⟨khối lệnh⟩. Lưu ý, ⟨biểu thức điều kiện⟩ phải được đặt trong cặp ngoặc

Kiểm tra một số có lớn hơn 6

#include <iostream>
using namespace std;

int main()
{
    float number;
    cout << "Nhap mot so trong khoang tu 1 den 10 = ";
    cin >> number;
    if (number > 6) {
        cout << "So ban nhap lon hon 6." << endl;
    }
    cout << number << " la so ban nhap.";
    return 0;
}

Câu lệnh if … else

if (⟨biểu thức điều kiện⟩)
{
    ⟨khối lệnh 1⟩
}
else
{
    ⟨khối lệnh 2⟩
}

Nếu ⟨biểu thức điều kiện⟩ cho kết quả đúng thì thực hiện ⟨khối lệnh 1⟩, ngược lại thì cho thực hiện ⟨khối lệnh 2⟩.

Giải và biện luận phương trình ax+b=0

#include <iostream>
using namespace std;

int main()
{
    float a, b;
    cout << "Nhap vao a: ";
    cin >> a;
    cout << "Nhap vao b: ";
    cin >> b;
    if (a == 0) {
        if (b == 0) {
            cout << "Phuong trinh vo so nghiem";
        }
        else {
            cout << "Phuong trinh vo nghiem";
        }
    }
    else {
        cout << "Nghiem phuong trinh x = " << -b / a;
    }
    return 0;
}

Câu lệnh switch

switch (⟨biểu thức số⟩)
{
    case ⟨n_1⟩: ⟨các câu lệnh 1⟩; break;
    case ⟨n_2⟩: ⟨các câu lệnh 2⟩; break;
    ...
    case ⟨n_k⟩: ⟨các câu lệnh k⟩; break;
    [default: ⟨các câu lệnh k+1⟩;]
}
  • Khi giá trị biểu thức bằng n_{i} thì thực hiện câu lệnh sau case ⟨n_i⟩.
  • Khi giá trị biểu thức không thỏa tất cả các n_{i} thì thực hiện câu lệnh sau default nếu có, hoặc thoát khỏi câu lệnh switch.
  • Khi chương trình đã thực hiện xong câu lệnh của case ⟨n_i⟩ nào đó thì nó sẽ thực hiện luôn các lệnh thuộc case bên dưới nó mà không xét lại điều kiện (do các n_i được xem như các nhãn) Vì vậy, để chương trình thoát khỏi lệnh switch sau khi thực hiện xong một trường hợp, ta dùng lệnh break.

Tạo thực đơn và cho phép chọn thực đơn bằng số nhập từ bàn phím.

#include <iostream>
using namespace std;

int main()
{
    int chon;
    cout << "Thuc Don" << endl;
    cout << "1. Lau thai!" << endl;
    cout << "2. Nuoc ngot!" << endl;
    cout << "3. Ca loc hap bau!" << endl;
    cout << "4. Chuot dong!" << endl;
    cout << "Xin moi ban chon mon an: ";
    cin >> chon;
    switch (chon)
    {
        case 1: cout << endl << "Ban chon lau thai!"; break;
        case 2: cout << endl << "Ban chon nuoc ngot!"; break;
        case 3: cout << endl << "Ban chon ca loc hap bau!"; break;
        case 4: cout << endl << "Ban chon chuot dong!"; break;
        default: cout << endl << "Ban chon khong dung!"; break;
    }
    return 0;
}

Cấu trúc lặp

Câu lệnh for

for (⟨biểu thức 1⟩; ⟨biểu thức 2⟩; ⟨biểu thức 3⟩)
{
    ⟨các câu lệnh⟩
}

Hoạt động của câu lệnh for như sau:

  • Bước 1: Khởi gán cho ⟨biểu thức 1⟩
  • Bước 2: Kiểm tra điều kiện của ⟨biểu thức 2⟩
    • Nếu ⟨biểu thức 2⟩ đúng thì cho thực hiện ⟨các câu lệnh⟩ và thực hiện ⟨biểu thức 3⟩ rồi quay trở lại bước 2.
    • Ngược lại thì thoát khỏi lặp.
  • Bất kỳ biểu thức nào trong 3 biểu thức nói trên đều có thể vắng nhưng phải giữ dấu chấm phẩy (;)

In ra màn hình bảng mã ASCII từ ký tự số 32 đến 255.

#include <iostream>
using namespace std;

int main()
{
    for (int i = 32; i <= 255; i++) {
        cout << "Ma ASCII cua " << (char)i << ": " << i << "\t";
    }
    return 0;
}

Câu lệnh while

while (⟨biểu thức điều kiện⟩)
{
    ⟨các câu lệnh⟩
}

Nếu ⟨biểu thức điều kiện⟩ đúng thì thực hiện ⟨các câu lệnh⟩ và lặp điều này cho đến khi nào ⟨biểu thức điều kiện⟩ sai thì kết thúc

In ra các chữ số của số nguyên n

#include <iostream>
using namespace std;

int main()
{
    int n, chuso;
    cout << "Nhap vao n = ";
    cin >> n;
    while (n > 0)
    {
        chuso = n % 10;
        cout << chuso << " ";
        n = n / 10;
    }
    return 0;
}

Câu lệnh do … while

do {
    ⟨các câu lệnh⟩
} while (⟨biểu thức điều kiện⟩);

Thực hiện khối lệnh cho đến khi ⟨biểu thức điều kiện⟩ có giá trị sai.

Nhập ký tự từ bàn phím hiển thị lên màn hình mã ASCII của ký tự đó, thực hiện đến khi nhấn phím ESC (Mã ASCII của phím ESC là 27).

#include <iostream>
#include <conio.h> // Lưu ý: conio.h không phải là thư viện chuẩn của C++
using namespace std;

int main()
{
    int ma;
    cout << "Nhan mot phim bat ky (Nhan ESC de thoat):" << endl;
    do {
        ma = getch();
        if (ma != 27) {
            cout << "Ma ASCII " << (char)ma << ": " << ma << "\t";
        }
    } while (ma != 27);
    return 0;
}

Lệnh lặp while kiểm tra điều kiện trước khi thực hiện lặp, còn lệnh lặp do...while thực hiện lệnh lặp rồi mới kiểm tra điều kiện. Do đó vòng lặp do...while thực hiện ⟨các câu lệnh⟩ ít nhất một lần.

Câu lệnh break và continue

Lệnh break dùng để kết thúc vòng lặp trực tiếp chứa nó

Cho phép người dùng nhập liên tục giá trị n cho đến khi nhập số âm thì dừng lại

#include <iostream>
using namespace std;

int main()
{
    int n;
    while (true)
    {
        cout << "Nhap n = ";
        cin >> n;
        if (n < 0) {
            break;
        }
    }
    return 0;
}

Lệnh continue dùng để bỏ qua một lần lặp.

In ra màn hình giá trị từ 1 đến 25 trừ đi số 14 và số 18.

#include <iostream>
using namespace std;

int main()
{
    for (int i = 1; i <= 25; i++)
    {
        if (i == 14 || i == 18) {
            continue;
        }
        cout << i << "\t";
    }
    return 0;
}

Cấu trúc nhảy

⟨nhãn⟩:
...
goto ⟨nhãn⟩;
...

con trỏ lệnh sẽ được nhảy đến vị trị của ⟨nhãn⟩

Kiểm tra một số nguyên là số dương, âm hay không?

#include <iostream>
using namespace std;

int main()
{
    int n;
    cout << "Nhap vao so nguyen = ";
    cin >> n;
    if (n > 0) goto nhan1;
    if (n < 0) goto nhan2;
    if (n == 0) goto nhan3;
nhan1:
    cout << "so duong";
    goto nhan;
nhan2:
    cout << "so am";
    goto nhan;
nhan3:
    cout << "so khong";
nhan:
    return 0;
}

Thực thi chương trình

  • Xác định các biến trong chương trình.
  • Giá trị ban đầu của mỗi biến.
  • Thực hiện chạy chương trình theo đúng trình tự đã được viết

Cho biết kết quả của đoạn chương trình sau:

#include <iostream>
using namespace std;

int main()
{
    int i, a = 4;
    for (i = 0; i < a; i++) {
        cout << i << endl;
    }
    return 0;
}

Cấu trúc rẽ nhánh

  1. Cho biết kết quả của đoạn chương trình sau:

    #include <iostream>
using namespace std;

int main() {
    int a = 9, b = 6;
    a++;
    a = a + b--;
    a = a + (--b);
    if (a % 2 == 0) {
        cout << "Gia tri cua a la chan";
    }
    cout << "Tong cua a va b la: " << a + b;
    return 0;
}

  2. Cho biết kết quả của đoạn chương trình sau:

    #include <iostream>
using namespace std;

int main() {
    int a = 7, b = 8;
    a++;
    a = a + (b--);
    --b;
    a--;
    a = (--a) + (--b);
    if (a % 2 != 0) {
        cout << endl << "a la so le";
    }
    else {
        cout << endl << "a la so chan";
    }
    cout << endl << "a = " << a;
    return 0;
}

  3. Cho biết kết quả của đoạn chương trình sau:

    #include <iostream>
using namespace std;

int main() {
    int x = 5, y;
    y = x++ + 5;
    cout << "x=" << x << ", y=" << y << endl;
    y *= 6;
    x = y % 7;
    cout << "x=" << x << ",y=" << y << ",y/x=" << y / x;
    return 0;
}

  4. Nhập vào hai số nguyên a,b. In ra màn hình giá trị lớn nhất.

  5. Cho ba số a,b,c đọc vào từ bàn phím. Hãy tìm giá trị lớn nhất của ba số trên và in ra kết quả.

  6. Cho ba số a,b,c đọc vào từ bàn phím. Hãy in ra màn hình theo thứ tự tăng dần các số. (Chỉ được dùng thêm hai biến phụ).

  7. Viết chương trình nhập vào một số nguyên n gồm ba chữ số. Xuất ra màn hình chữ số lớn nhất ở vị trí nào? Ví dụ: n=291. Chữ số lớn nhất nằm ở hàng chục là 9.

  8. Viết chương trình nhập vào số nguyên n gồm ba chữ số. Xuất ra màn hình theo thứ tự tăng dần của các chữ số. Ví dụ: n=291. Xuất ra 129.

  9. Nhập vào ngày, tháng, năm. Kiểm tra xem ngày, tháng, năm đó có hợp lệ hay không? In kết quả ra màn hình.

  10. Nhập vào giờ, phút, giây. Kiểm tra xem giờ, phút, giây đó có hợp lệ hay không? In kết quả ra màn hình.

  11. Viết chương trình nhập vào ngày, tháng, năm hợp lệ. Cho biết năm này có phải là năm nhuận hay không? In kết quả ra màn hình.

  12. Viết chương trình tính diện tích và chu vi các hình: tam giác, hình vuông, hình chữ nhật và hình tròn với những thông tin cần được nhập từ bàn phím.

  13. Viết chương trình tính tiền cước TAXI. Biết rằng:

    1. km đầu tiên giá là 5000đ

    2. 200m tiếp theo là 1000đ.

    3. Nếu lớn hơn 30km thì mỗi km thêm sẽ là 3000đ

3.3 TÓM TẮT

Cấu trúc tuần tự, lặp và rẽ nhánh (lựa chọn) là ba cấu trúc chính hình thành nên chương trình. Dựa vào những cấu trúc điều khiển này ta có thể xây dựng thành những chương trình phức tạp hơn. Vì vậy phải nắm rõ cách hoạt động của những cấu trúc điều khiển này để cài đặt đúng yêu cầu bài toán. Khi sử dụng phải lưu ý điều kiện thực hiện hay kết thúc của một thao tác nào đó. Bên trong một phát biểu điều khiển phải là một lệnh hay một khối lệnh (khối lệnh được đặt bên trong cặp dấu ngoặc {}). Những biến không phụ thuộc vào vòng lặp nên đặt bên ngoài vòng lặp. Khi sử dụng cấu trúc điều khiển lồng nhau phải lưu ý vị trí mở ngoặc hay đóng ngoặc cho hợp lý.
